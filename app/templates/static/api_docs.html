⚡ Быстрый старт (curl)

# 1. Получить токен
curl -X POST /api/v1/auth/login \
  -H "Content-Type: application/json" \
  -d '{"login":"admin","password":"Admin123!"}'

# 2. Получить список видео
curl /api/v1/videos?sort=popular&per_page=5

# 3. Поиск
curl "/api/v1/search?q=minecraft"

# 4. Защищённый запрос (лайк)
curl -X POST /api/v1/videos/1/like \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"is_like":true}'
